친선경기 리셋 코드


(function() {
    const myId = (typeof currentUser === 'string' && currentUser) ? currentUser : "ooks";
    
    // 1. 전역 상태 횟수 및 대결 인덱스 초기화
    friendlyMatchesToday = 0;
    friendlyCurrentOpponentIndex = 0;
    
    // 2. 로컬 스토리지 데이터 0으로 갱신
    localStorage.setItem(`fc_star_friendly_matches_today_${myId}`, '0');
    localStorage.setItem(`fc_star_friendly_current_index_${myId}`, '0');
    
    // 3. UI 및 경기 프리뷰 즉시 갱신
    if (typeof updateFriendlyMatchPreview === 'function') updateFriendlyMatchPreview();
    if (typeof renderFriendlyTable === 'function') renderFriendlyTable();
    
    // 4. 리셋 상태 클라우드 Firestore에 동기화 백업
    if (typeof saveUserProgress === 'function') saveUserProgress();
    
    // 5. 알림 출력
    if (typeof showToast === 'function') {
        showToast("🔓 친선경기 잔여 횟수 및 대결 순서가 리셋되었습니다! (3/3)");
    } else {
        console.log("🟢 친선경기 리셋 완료! (잔여 횟수: 3/3)");
    }
})();


일반등급으로 다시 돌아가는 콘솔코드를 알려줘


(function() {
    // 1. 어려움 모드 전역 상태 해제 및 로컬 저장소 값 변경
    isHardMode = false;
    localStorage.setItem('fc_star_is_hard_mode', 'false');
    
    // 2. 인게임 엠블럼 후광(Glow) 테마 및 붉은색 보더 제거
    if (typeof updateGlowTheme === 'function') {
        updateGlowTheme();
    }
    
    // 3. 리그 프리뷰 보드 내 어려움 모드 뱃지(Indicator) 숨기기
    const hardModeIndicator = document.getElementById('leagueHardModeIndicator');
    if (hardModeIndicator) {
        hardModeIndicator.style.display = 'none';
    }
    
    // 4. UI 및 스쿼드, OVR 관련 수치 즉시 갱신
    if (typeof renderUserPoints === 'function') renderUserPoints();
    if (typeof renderDeck === 'function') renderDeck();
    if (typeof renderSquadFormation === 'function') renderSquadFormation();
    if (typeof syncJeonbukOvr === 'function') syncJeonbukOvr();
    if (typeof updateMatchPreviewBoard === 'function') updateMatchPreviewBoard();
    if (typeof renderLeagueTable === 'function') renderLeagueTable();
    
    // 5. 변경된 일반 모드 상태를 Firestore 클라우드 원격 서버에 즉시 백업 저장
    if (typeof saveUserProgress === 'function') {
        saveUserProgress();
    }
    
    // 6. 피드백 알림 출력
    if (typeof showToast === 'function') {
        showToast("🔓 일반 모드로 정상 전환되었습니다!");
    } else {
        alert("일반 모드로 정상 전환되었습니다!");
    }
})();
